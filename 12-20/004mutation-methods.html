			Vue.js 包装了被观察数组的变异方法，故它们能触发视图更新。被包装的方法有：
			push()/*从数组的尾部加入新的，并返回新数组长度*/
			test.items.push({mes:"zhou"})
			pop()/*从数组尾部弹出*/
			test.items.pop()
			shift()/*从数组头部弹出*/
			test.items.shift()
			unshift()/*从数组头部插入*/
			test.items.unshift({mes:"bai"})
			splice()/*如js中的方法，增加，删除*/
			test.items.splice()
			sort()/*排序*/
			test.items.sort()
			reverse()/*反转数组*/
			test.items.reverse()
